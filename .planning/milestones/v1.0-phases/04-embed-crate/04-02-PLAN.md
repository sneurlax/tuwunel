```rust
pub async fn stop(self) -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
    // Signal shutdown via broadcast channel
    self.server.shutdown()?;

    // Wait for the run task to finish
    // run() returns after receiving the shutdown signal
    self.run_handle.await??;

    // Stop services (cleanup DB references, etc.)
    tuwunel_router::stop(self.services).await?;

    // _db_dir (TempDir) drops here, cleaning up RocksDB tempdir
    // _flame_guard drops here
    Ok(())
}
```

**2. Implement `register_user()`:**
Per D-03 (EMBD-09). Uses standalone reqwest calls with the UIAA two-step flow, same pattern as `tests/shadow/src/scenarios/common.rs::MatrixClient::register_with_token()` but self-contained.

```rust
pub async fn register_user(
    &self,
    username: &str,
    password: &str,
    registration_token: &str,
) -> Result<RegisteredUser, Box<dyn std::error::Error + Send + Sync>> {
    let client = reqwest::Client::builder()
        .timeout(std::time::Duration::from_secs(10))
        .build()?;

    let url = format!("{}/_matrix/client/v3/register", self.base_url);

    // Step 1: initial registration attempt to get UIAA session
    let body = serde_json::json!({
        "username": username,
        "password": password,
        "auth": { "type": "m.login.dummy" }
    });

    let resp = client.post(&url).json(&body).send().await?;
    let status = resp.status();
    let resp_body: serde_json::Value = resp.json().await?;

    // If succeeded directly (unlikely with registration token enabled)
    if status.is_success() {
        let access_token = resp_body
            .get("access_token")
            .and_then(|v| v.as_str())
            .unwrap_or_default()
            .to_owned();
        let user_id = resp_body
            .get("user_id")
            .and_then(|v| v.as_str())
            .unwrap_or_default()
            .to_owned();
        return Ok(RegisteredUser { user_id, access_token });
    }

    // Extract session from UIAA 401 response
    let session = resp_body
        .get("session")
        .and_then(|s| s.as_str())
        .ok_or_else(|| format!(
            "Registration UIAA response missing session for {username}: {resp_body}"
        ))?;

    // Step 2: retry with registration token
    let body = serde_json::json!({
        "username": username,
        "password": password,
        "auth": {
            "type": "m.login.registration_token",
            "token": registration_token,
            "session": session
        }
    });

    let resp = client.post(&url).json(&body).send().await?;
    let status = resp.status();
    let resp_body: serde_json::Value = resp.json().await?;

    if !status.is_success() {
        return Err(format!(
            "Registration failed for {username}: {status} {resp_body}"
        ).into());
    }

    let access_token = resp_body
        .get("access_token")
        .and_then(|v| v.as_str())
        .unwrap_or_default()
        .to_owned();
    let user_id = resp_body
        .get("user_id")
        .and_then(|v| v.as_str())
        .unwrap_or_default()
        .to_owned();

    Ok(RegisteredUser { user_id, access_token })
}
```
